- 항상 명령어 마지막에 세미콜론(;)입력 
- 출력 함수는 print()

```C#
using UnityEngine;

// MonoBehaviour(부모 클래스) 속성을 상속받은 클래스
public class Study : MonoBehaviour 
{
    void Start() 
    {
        // 안녕하세요~! 출력
        print("안녕하세요~!");
    }
} 
```
